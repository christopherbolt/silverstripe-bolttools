### Chris Bolt Begin ###

# Block bad bots
RewriteCond %{HTTP_USER_AGENT} Bytespider [NC]
RewriteRule .* - [F,L] 

# Security
Header always set Strict-Transport-Security "max-age=31536000; includeSubDomains; preload"
Header always set Content-Security-Policy "frame-ancestors 'self'"
Header always set X-Frame-Options "SAMEORIGIN"

# Caching
# 1 WEEK
<IfModule mod_headers.c>
	<FilesMatch "\.(jpg|jpeg|jpgx|jpegx|png|gif|webp|avif|swf|pdf|js|css|html|eot|svg|ttf|woff|woff2)$">
		Header set Cache-Control "max-age=31536000, public"
	</FilesMatch>
</IfModule>
# End Caching

# Gzip compress javascript and css files
<IfModule mod_deflate.c>
	AddType image/svg+xml .svg
	AddOutputFilterByType DEFLATE text/css application/x-javascript application/javascript text/javascript image/svg+xml
</IfModule>
# End Gzip

# Hide git
RedirectMatch 404 /\.git
RedirectMatch 404 /*\.md

# Redirect themes
RewriteRule ^themes/(.*) resources/themes/$1 [NC,L]
